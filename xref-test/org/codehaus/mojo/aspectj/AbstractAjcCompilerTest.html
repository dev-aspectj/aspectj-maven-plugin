100     /**
101      * Tests if modules told to weave that are found in the project dependencies actually are found in the .inpath ajc
102      * argument,.
103      *
104      * @throws MojoExecutionException if the mojo fails to execute
105      */
106     public void testGetAjcArguments_weaveArtifacts() throws MojoExecutionException
107     {
108         ajcCompMojo.weaveDependencies = new Module[2];
109         Module module1 = new Module();
110         String mod1Group = "dill.group";
111         module1.setGroupId( mod1Group );
112         String mod1Artifact = "dall.artifact";
113         module1.setArtifactId( mod1Artifact );
114         ajcCompMojo.weaveDependencies[0] = module1;
115         Module module2 = new Module();
116         String mod2Group = "foooup";
117         module2.setGroupId( mod2Group );
118         String mod2Artifact = "bartifact";
119         module2.setArtifactId( mod2Artifact );
120         ajcCompMojo.weaveDependencies[1] = module2;
121         // Modify project to include depencies
122         Set artifacts = new HashSet();
123         artifacts.add( new MockArtifact( mod1Group, mod1Artifact ) );
124         artifacts.add( new MockArtifact( mod2Group, mod2Artifact ) );
125         ajcCompMojo.project.setArtifacts( artifacts );
126         ajcCompMojo.assembleArguments();
127         List args = ajcCompMojo.ajcOptions;
128         assertTrue( args.contains( "-inpath" ) );
129         Iterator it = args.iterator();
130         while ( !it.next().equals( "-inpath" ) )
131         {
132             // don't do nothing
133         }
134         String weavePath = (String) it.next();
135         assertTrue( weavePath.indexOf( File.pathSeparator ) != -1 );
136         assertTrue( weavePath.indexOf( mod1Artifact ) != -1 );
137         assertTrue( weavePath.indexOf( mod2Artifact ) != -1 );
138     }
